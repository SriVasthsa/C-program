void rev string(char str[])
{
int start= 0;

{

5.Pointer is storing the address of another variable

#include<stdio.h>
void swap(int*a, int*b)
{
int temp=*a;
*a=*b;
*b=temp;
}
int main()
{
int x= 10;
int y=30;
printf("\n");
printf(x=%d,y=%d\n", x, y);
swap(&x,&y);
printf(" "\n");
printf(x=%d, y=%d\n", x, y;
return 0;
}
